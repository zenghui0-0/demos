[remote]
1, check: ps -e|grep ssh 
2, install: apt-get install ssh
3, start: sudo /etc/init.d/ssh start
4, enable root remote: sudo vim /etc/ssh/sshd_config; PermitRootLogin yes
5, restart: service ssh restart
